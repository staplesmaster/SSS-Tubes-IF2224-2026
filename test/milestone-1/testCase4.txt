type Point(
    x, y: integer;
) 

record(
    employee, (id, name);
)

var:
    now, ends: Point;
    i: integer;

begin
    now := Point(0, 0);
    ends := Point(10, 10);
    for i := 10 downto 0 do
        case i mod 2 == 0:
            now.x := now.x + 1;
            now.y := now.y + 1;
        case i mod 2 <> 1:
            ends.x := end.x - 1; 
            ends.y := end.y - 1; 

    for i := 0 to 10 do
        case i mod 2 <> 0:
            now.x := now.x + 1;
            now.y := now.y + 1;
        case i mod 2 == 1:
            ends.x := end.x - 1; 
            ends.y := end.y - 1; 
    print('Point = ', x, 'And', y);
end.
